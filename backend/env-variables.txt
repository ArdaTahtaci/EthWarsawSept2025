# GolemDB Configuration Environment Variables
# Copy these to your .env file and fill in the appropriate values

# Basic server configuration
PORT=3000

# GolemDB Chain Configuration
# Chain ID for the blockchain network (1 = Ethereum Mainnet, 11155111 = Sepolia, etc.)
GOLEMDB_CHAIN_ID=1

# REQUIRED: RPC URL for blockchain connection
# You can use public RPCs or your own node
GOLEMDB_RPC_URL=https://eth.llamarpc.com

# Optional: WebSocket URL for real-time updates
GOLEMDB_WS_URL=wss://eth.llamarpc.com

# Account Data (for write operations)
# This should be your private key or account configuration
# Leave empty for read-only access
GOLEMDB_ACCOUNT_DATA=your_account_private_key_here

# GolemDB Read-Only Mode
# Set to "true" to force read-only mode even if account data is provided
# Set to "false" or leave empty to use full client when account data is available
GOLEMDB_READ_ONLY=false

# Example configuration for Ethereum Mainnet (read-only):
# GOLEMDB_CHAIN_ID=1
# GOLEMDB_RPC_URL=https://eth.llamarpc.com
# GOLEMDB_WS_URL=wss://eth.llamarpc.com
# GOLEMDB_READ_ONLY=true

# Example configuration for Sepolia Testnet (read-only):
# GOLEMDB_CHAIN_ID=11155111
# GOLEMDB_RPC_URL=https://sepolia.llamarpc.com
# GOLEMDB_WS_URL=wss://sepolia.llamarpc.com
# GOLEMDB_READ_ONLY=true

# Example configuration with write access (requires private key):
# GOLEMDB_CHAIN_ID=1
# GOLEMDB_RPC_URL=https://eth.llamarpc.com
# GOLEMDB_WS_URL=wss://eth.llamarpc.com
# GOLEMDB_ACCOUNT_DATA=0x1234567890abcdef...
# GOLEMDB_READ_ONLY=false

# Additional Notes:
# - Use a reliable RPC provider (Infura, Alchemy, QuickNode, or public RPCs)
# - For write operations, you need a private key with sufficient ETH for gas fees
# - Keep your private key secure and never commit to version control
# - Use testnet chains (Sepolia, Goerli) for development and mainnet for production
# - WebSocket URL is optional but recommended for real-time updates
# - Read-only mode doesn't require any account data or private keys
